/**
 * This list contains all methods for using the yfinance library from Yahoo. 
 * Please use responsible, If your IP gets flag for unusual activities, thats on you!
 * not on us. Read the rest of the documentation
 */

/**
 * Retrieves the current price of the stock.
 */
get_price()

/**
 * Retrieves the change in price of the stock.
 */
get_change()

/**
 * Retrieves the percentage change in price of the stock.
 */
get_percent_change()

/**
 * Retrieves the trading volume of the stock.
 */
get_volume()

/**
 * Retrieves the previous closing price of the stock.
 */
get_prev_close()

/**
 * Retrieves the opening price of the stock.
 */
get_open()

/**
 * Retrieves the average daily trading volume of the stock.
 */
get_avg_daily_volume()

/**
 * Retrieves the stock exchange where the stock is listed.
 */
get_stock_exchange()

/**
 * Retrieves the market capitalization of the stock.
 */
get_market_cap()

/**
 * Retrieves the book value of the stock.
 */
get_book_value()

/**
 * Retrieves the EBITDA (Earnings Before Interest, Taxes, Depreciation, and Amortization) of the stock.
 */
get_ebitda()

/**
 * Retrieves the dividend per share of the stock.
 */
get_dividend_share()

/**
 * Retrieves the dividend yield of the stock.
 */
get_dividend_yield()

/**
 * Retrieves the earnings per share of the stock.
 */
get_earnings_share()

/**
 * Retrieves the highest price of the stock during the trading day.
 */
get_days_high()

/**
 * Retrieves the lowest price of the stock during the trading day.
 */
get_days_low()

/**
 * Retrieves the highest price of the stock in the past year.
 */
get_year_high()

/**
 * Retrieves the lowest price of the stock in the past year.
 */
get_year_low()

/**
 * Retrieves the 50-day moving average price of the stock.
 */
get_50day_moving_avg()

/**
 * Retrieves the 200-day moving average price of the stock.
 */
get_200day_moving_avg()

/**
 * Retrieves the price-to-earnings ratio of the stock.
 */
get_price_earnings_ratio()

/**
 * Retrieves the price-to-earnings growth ratio of the stock.
 */
get_price_earnings_growth_ratio()

/**
 * Retrieves the price-to-sales ratio of the stock.
 */
get_price_sales()

/**
 * Retrieves the price-to-book ratio of the stock.
 */
get_price_book()

/**
 * Retrieves the short interest ratio of the stock.
 */
get_short_ratio()

/**
 * Retrieves the date and time of the last trade of the stock.
 */
get_trade_datetime()

/**
 * Retrieves the historical stock data within a specified date range.
 * @param {string} start_date - The start date of the historical data.
 * @param {string} end_date - The end date of the historical data.
 */
get_historical(start_date, end_date)

/**
 * Retrieves additional information about the stock.
 */
get_info()

/**
 * Retrieves the name of the stock.
 */
get_name()

/**
 * Refreshes the stock data.
 */
refresh()

/**
 * Retrieves the percentage change from the year high price of the stock.
 */
get_percent_change_from_year_high()

/**
 * Retrieves the percentage change from the year low price of the stock.
 */
get_percent_change_from_year_low()

/**
 * Retrieves the change from the year low price of the stock.
 */
get_change_from_year_low()

/**
 * Retrieves the change from the year high price of the stock.
 */
get_change_from_year_high()

/**
 * Retrieves the percentage change from the 200-day moving average price of the stock.
 */
get_percent_change_from_200_day_moving_average()

/**
 * Retrieves the change from the 200-day moving average price of the stock.
 */
get_change_from_200_day_moving_average()

/**
 * Retrieves the percentage change from the 50-day moving average price of the stock.
 */
get_percent_change_from_50_day_moving_average()

/**
 * Retrieves the change from the 50-day moving average price of the stock.
 */
get_change_from_50_day_moving_average()

/**
 * Retrieves the estimated earnings per share for the next quarter.
 */
get_EPS_estimate_next_quarter()

/**
 * Retrieves the estimated earnings per share for the next year.
 */
get_EPS_estimate_next_year()

/**
 * Retrieves the ex-dividend date of the stock.
 */
get_ex_dividend_date()

/**
 * Retrieves the estimated earnings per share for the current year.
 */
get_EPS_estimate_current_year()

/**
 * Retrieves the price-to-earnings ratio based on the estimated earnings per share for the next year.
 */
get_price_EPS_estimate_next_year()

/**
 * Retrieves the price-to-earnings ratio based on the estimated earnings per share for the current year.
 */
get_price_EPS_estimate_current_year()

/**
 * Retrieves the one-year target price of the stock.
 */
get_one_yr_target_price()

/**
 * Retrieves the change and percentage change in price of the stock.
 */
get_change_percent_change()

/**
 * Retrieves the dividend pay date of the stock.
 */
get_dividend_pay_date()

/**
 * Retrieves the currency in which the stock is traded.
 */
get_currency()

/**
 * Retrieves the last trade price of the stock along with the date and time.
 */
get_last_trade_with_time()

/**
 * Retrieves the range of prices of the stock during the trading day.
 */
get_days_range()

/**
 * Retrieves the range of prices of the stock in the past year.
 */
get_year_range()
